384  /**
385  * Somehow inspired by ezPODocScanner::objInspect()
386  * Lists the publicly accessible properties of a class as an array name => type
387  * NB: returns php types, not xsd ones
388  * @todo move to another class?
389  */
390  static function classInspect( $classname )
391  {
392  if ( !class_exists( $classname ) )
393  {
394  eZDebug::writeError( "Cannot introspect class $classname for wsdl generation: class not found", __METHOD__ );
395  return array();
396  }
397  $classmethods = get_class_methods( $classname );
398  if ( in_array( "attributes", $classmethods ) && in_array( "attribute", $classmethods ) )
399  {
400  // 'template object' (should be a descendant of ezpo)
401  $out = array();
402  if ( is_callable( 'ezPODocScanner::definition' ) )
403  {
404  // load theorical desc parsed from online docs. A warning is logged by ezPODocScanner if class is not found
405  $defs = ezPODocScanner::definition( strtolower( $classname ) );
406  if ( isset( $defs['attributes'] ) )
407  {
408  foreach( $defs['attributes'] as $name => $attr )
409  {
410  if ( strpos( $attr['type'], 'object [' ) === 0 )
411  {
412  $typename = substr( $attr['type'], 8, -1 );
413  // currently we get lowercase class names, but we want correct casing
414  // (add strtolower in case in the future ezPODocScanner is fixed)
415  $typename = ezPODocScanner::findClassNameGivenLowerCaseName( strtolower( $typename ) );
416  $out[$name] = $typename;
417  }
418  else if ( strpos( $attr['type'], 'array [' ) === 0 )
419  {
420  $typename = substr( $attr['type'], 7, -1 );
421  if ( !ezPODocScanner::isscalar( $typename ) && $typename != 'array' )
422  {
423  $typename = ezPODocScanner::findClassNameGivenLowerCaseName( strtolower( $typename ) );
424  }
425  $out[$name] = 'array of ' . $typename;
426  }
427  else
428  {
429  $out[$name] = $attr['type'];
430  }
431  }
432  }
433  }
434  else
435  {
436  eZDebug::writeError( "Cannot introspect subclass of eZPO for wsdl generation ($classname): missing extension ezpersistentobject_inspector", __METHOD__ );
437  }
438  return $out;
439  }
440  else
441  {
442  // not a template class: do a dump based on reflection
443  $out = array();
444  $reflectionClass = new ReflectionClass( $classname );
445  foreach( $reflectionClass->getProperties( ReflectionProperty::IS_PUBLIC | ReflectionProperty::IS_STATIC ) as $prop )
446  {
447  $doc = $prop->getDocComment();
448  $type = $doc == '' ? 'mixed' : self::typeFromDocComment( $doc );
449  $out[$prop->name] = $type;
450  }
451  return $out;
452  }
453  }

471  * Used by the permission system: check if current user has access to ws method
472  * @param string $functionName
473  * @param ezuser $user
474  */
475  static function checkAccess( $functionName, $user=null )
476  {
477  if ( $user == null )
478  {
479  $user = eZUser::currentUser();
480  }
481 
482  $access = false;
483  $accessResult = $user->hasAccessTo( 'webservices' , 'execute' );
484  $accessWord = $accessResult['accessWord'];
485  if ( $accessWord == 'yes' )
486  {
487  $access = true;
488  }
489  else if ( $accessWord != 'no' ) // with limitation
490  {
491  $currentsa = eZSys::ezcrc32( $GLOBALS['eZCurrentAccess']['name'] );
492  $functionName = md5( $functionName );
493  $accessws = 1;
494  $accesssa = 1;
495  foreach ( $accessResult['policies'] as $key => $policy )
496  {
497  if ( isset( $policy['Webservices'] ) && $accessws === 1 )
498  {
499  $accessws = false;
500  }
501  if ( isset( $policy['Webservices'] ) && in_array( $functionName, $policy['Webservices'] ) )
502  {
503  $accessws = true;
504  }
505  if ( isset( $policy['SiteAccess'] ) && $accesssa === 1 )
506  {
507  $accesssa = false;
508  }
509  if ( isset( $policy['SiteAccess'] ) && in_array( $currentsa, $policy['SiteAccess'] ) )
510  {
511  $accesssa = true;
512  }
513  }
514  $access = $accessws && $accesssa;
515  }
516 
517  return $access;
518  }
